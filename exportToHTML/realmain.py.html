508      def fft(self, array): 
509          yf = fft(array, n=self.nfft) 
510          xf = np.linspace(0.0, self.samplingRate / 2, self.nfft / 2) 
511          return xf, abs(yf[0:self.nfft / 2])

513      def HanningFunction(radarData): 
514          return radarData * np.hanning(len(radarData))

516      def getcorrMtrx(self, x, m): 
517          x = np.array(x) 
518          m = m 
519          N = len(x) 
520          xlen = m + 1 
521          rowVector = x[N - xlen: N] 
522          columnVector = x[0: N - m] 
523          hanMatrix = hankel(c=np.array(columnVector).T, r=rowVector) 
524          X_unscaled = np.fliplr(hanMatrix) 
525          X = X_unscaled / np.sqrt(N - m) 
526          Xnew = np.conj(np.fliplr(X)) 
527          corrMatrix = np.vstack((X, Xnew)) / np.sqrt(2) 
528          return corrMatrix

530      def musicAlg(self, corrMtrx): 
531          u, s, v = np.linalg.svd(corrMtrx) 
532   
533          self.nfft = 512 
534   
535          frequencyVector = np.linspace(0, 1, self.nfft // 2) 
536          frequencyVector *= self.samplingRate / 2 
537   
538          sum = 0 
539          for i in range(self.p, len(v)): 
540              y = fft(v[i], n=self.nfft) 
541              sum += abs(y) ** 2 / s[i] 
542   
543          sum = 1 / sum 
544   
545          sum = sum[0:self.nfft // 2] 
546   
547          return frequencyVector, sum, s

578      def butter_lowpass(cutoff, fs, order=5): 
579          nyq = 0.5 * fs 
580          normal_cutoff = cutoff / nyq 
581          b, a = butter(order, normal_cutoff, btype='low', analog=False) 
582          return b, a

584      def butter_lowpass_filter(data, cutoff, fs, order=5): 
585          b, a = Radar.butter_lowpass(cutoff, fs, order=order) 
586          y = lfilter(b, a, data) 
587          return y
